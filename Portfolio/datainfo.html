Query Snippet

SELECT DISTINCT COUNT(actor.actnumb) as 'Actors in action movies after 2010'
FROM actor inner join role
ON actor.actnumb = role.actnumb inner join movie
ON role.mvnumb = movie.mvnumb
WHERE mvtype = 'Action'and yearmade > 2010

SELECT * FROM movie inner join director
ON movie.dirnumb = director.dirnumb inner join role
ON movie.mvnumb = role.mvnumb inner join actor
ON role.actnumb = actor.actnumb
WHERE dirname = 'Lana Wachowski'and actname = 'Keanu Reeves'
ORDER BY movie.mvtitle
